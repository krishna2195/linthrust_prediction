import pandas as pd
import numpy as np
from sklearn.decomposition import PCA
import statsmodels.api as sm
from sklearn.preprocessing import StandardScaler

# Standardize the predictor variables (important for PCA)
scaler = StandardScaler()
X_standardized = scaler.fit_transform(X_1)

# Perform PCA
pca = PCA()
X_pca = pca.fit_transform(X_standardized)
#print(X_pca)
# Determine the number of principal components to include based on explained variance
explained_variance_ratio = pca.explained_variance_ratio_
cumulative_explained_variance = np.cumsum(explained_variance_ratio)

# Determine the number of components that explain a certain percentage of variance (e.g., 95%)
target_explained_variance = 0.99
num_components = np.argmax(cumulative_explained_variance >= target_explained_variance) + 1
#num_components=11

# Print the number of selected principal components
print(f"\nNumber of Selected Principal Components: {num_components} \n\n")

# Retain only the selected number of principal components
X_pca_selected = X_pca[:, :num_components]

# Add a constant term to the principal components
X_pca_selected = sm.add_constant(X_pca_selected)

# Fit the ordinary least squares (OLS) model using PCR components
model_pcr = sm.OLS(y_1, X_pca_selected).fit()

# Display PCR regression summary
print(model_pcr.summary())


# Get the loadings from the PCA
loadings = pca.components_[:num_components, :].T

# Extract the coefficients from the PCR model
coefficients_pcr = model_pcr.params.values[1:]

# Include the intercept separately
intercept_pcr = model_pcr.params.values[0]

# Calculate the coefficients for the original variables
coefficients_original = loadings @ coefficients_pcr

# Add the intercept to the coefficients for the original variables
coefficients_original_with_intercept = np.insert(coefficients_original, 0, intercept_pcr)


# Display the coefficients for the original variables
coefficients_data = pd.DataFrame({'Variable': X_1.columns, 'Coefficient': coefficients_original})
print("\nCoefficients for Original Variables:")
print(coefficients_data)

# Stepwise Regression with both forward and backward steps
def stepwise_selection(X, y, alpha=0.1):
    included = []
    print("\nStepwise Regression Results:")
    
    while True:
        changed = False

        # Forward step
        excluded = list(set(X.columns) - set(included))
        new_pval = pd.Series(index=excluded)

        for new_column in excluded:
            model = sm.OLS(y, sm.add_constant(X[included + [new_column]])).fit()
            new_pval[new_column] = model.pvalues[new_column]

        best_column = new_pval.idxmin()
        if new_pval[best_column] < alpha:
            included.append(best_column)
            print(f"Forward Step: Added {best_column}. P-value: {new_pval[best_column]:.4f}")
            changed = True

        # Backward step
        model = sm.OLS(y, sm.add_constant(X[included])).fit()
        pvalues = model.pvalues
        worst_column = pvalues.idxmax()
        if worst_column in included and pvalues[worst_column] >= alpha:
            included.remove(worst_column)
            print(f"Backward Step: Removed {worst_column}. P-value: {pvalues[worst_column]:.4f}")
            changed = True

        if not changed:
            break
        
    print("\nFinal Included Variables:")
    print(included)
    return included

# Perform stepwise regression
selected_columns = stepwise_selection(X_2, y_2)

import pandas as pd
import numpy as np
from itertools import combinations
from statsmodels.api import OLS, add_constant
from statsmodels.stats.outliers_influence import variance_inflation_factor
from sklearn.model_selection import train_test_split

# Load the data
data = pd.read_csv('c:/Drive_d/Applied Stats/project/LINTH-5.txt', sep='\s+')

# Extract response variable and predictor variables
y = data['BIO']
X = data[['SAL', 'pH', 'K', 'Na', 'Zn']]

# Function to calculate BIC for a given model
def calculate_bic(model, X, y):
    n = len(y)
    p = model.df_model + 1  # Add 1 for the intercept
    bic = n * np.log(np.sum((model.predict(X) - y)**2) / n) + p * np.log(n)
    return bic

# Function to calculate VIF for a given model
def calculate_vif(X):
    vif_data = pd.DataFrame()
    vif_data["Variable"] = X.columns
    vif_data["VIF"] = [variance_inflation_factor(X.values, i) for i in range(X.shape[1])]
    return vif_data

# Function for subset selection based on BIC and VIF
def subset_selection(X, y, max_vars=2):
    best_bic = np.inf
    best_vif = np.inf
    best_model = None

    for k in range(1, max_vars + 1):
        # Generate all combinations of k variables
        variable_combinations = list(combinations(X.columns, k))

        for variables in variable_combinations:
            # Fit a model using the selected variables
            X_subset = X[list(variables)]
            X_subset = add_constant(X_subset)
            model = OLS(y, X_subset).fit()

            # Calculate BIC for the model
            bic = calculate_bic(model, X_subset, y)
            
            # Calculate VIF for the model
            vif_values = calculate_vif(X_subset)

           # If BIC is better than the current best, or if there's a tie in BIC, compare VIF
            if bic < best_bic or (bic == best_bic and max(vif_values) < best_vif):
                best_bic = bic
                best_vif = max(vif_values)
                best_model = model

    return best_model

# Perform subset selection
best_subset_model = subset_selection(X, y, max_vars=2)

# Display results
print("Best Subset Model:")
print(best_subset_model.summary())

# Calculate VIF for the selected variables
vif_data_subset = calculate_vif(add_constant(X[list(best_subset_model.params.index[1:])]))

# Display VIF results
print("\nVariance Inflation Factor (VIF) for Selected Variables:")
print(vif_data_subset)
